
# Create Work Directory
mqsicreateworkdir /var/ace/myNode

# Create Integration Node
mqsicreatebroker myNode

# Start Integration Node
mqsistart myNode

# Stop Integration Node
mqsistop myNode

# Delete Integration Node
mqsiremovebroker myNode

=============================

# Create an Integration Server under Node
mqsicreateexecutiongroup myNode -e APP_SERVER

# Restart Integration Server
mqsireload myNode -e APP_SERVER

# List all Integration Servers
mqsilist myNode
